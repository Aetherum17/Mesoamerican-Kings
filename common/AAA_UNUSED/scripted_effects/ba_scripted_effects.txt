ba_impact_from_unkilled_deformed_effect = { # Root is the character and CHILD is the scope (like scope:child)
	if = {
		limit = {
			faith = { has_doctrine_parameter = infanticide_any }
			$CHILD$ = {
				ba_has_deformity_trigger = yes
			}
		}
		add_piety = didnt_kill_deformed_piety_loss
		add_piety_level = -1
		stress_impact = {
			base = minor_stress_impact_gain
			zealous = major_stress_impact_gain
			callous = major_stress_impact_gain
			sadistic = massive_stress_impact_gain
			compassionate = major_stress_impact_loss
		}
	}
}
ba_impact_from_infanticide_effect = { # Same as above
	if = {
		limit = {
			faith = { has_doctrine_parameter = infanticide_any }
			$CHILD$ = {
				ba_has_deformity_trigger = yes
			}
		}
		add_piety = killed_deformed_piety_gain
		stress_impact = {
			compassionate = major_stress_impact_gain
		}
	}
	else = {
		add_piety = killed_nondeformed_piety_gain
		stress_impact = {
			base = minor_stress_gain
			compassionate = massive_stress_impact_gain
		}
	}
}
ba_culture_exposure_effect = { #NAME is the name in innovation_NAME and the variable NAME_exposure
	# CULTURE_TAKER is the culture recieving the innovation, CULTURE_GIVER gives it
	if = {
		limit = { # Holder gives the subject exposure to innovation
			$CULTURE_GIVER$ = { has_innovation = innovation_$NAME$ }
			$CULTURE_TAKER$ = { NOT = { has_variable = $NAME$_exposure } }
		}
		$CULTURE_TAKER$ = {
			set_variable = {
				name = $NAME$_exposure
				value = 1
			}
		}
		#Player Alerts
		$CULTURE_TAKER$ = { save_scope_as = culture_reciever }
		every_player = {
			limit = {
				culture = scope:culture_reciever
			}
			trigger_event = {
				id = $NAME$_alert.0001
			}
		}
	}
}
ba_add_innovation_with_exposure_effect = { #Start scope is culture and NAME is the innovation_NAME
	set_variable = {
		name = $NAME$_exposure
		value = 1
	}
	add_innovation = innovation_$NAME$
}

ba_found_dynasty_effect = {
	custom_tooltip = create_new_kingdom

	hidden_effect = {
		save_scope_as = founder
		primary_title = {
			save_scope_as = old_title
		}
		
		create_dynamic_title = {
			tier = kingdom
			name = NEW_CREATED_TITLE_NAME
		}
		create_title_and_vassal_change = change
		scope:change = {
			set_title_and_vassal_change_type = created
		}
		
		scope:new_title = {
			change_title_holder = {
				holder = root
				change = scope:change
			}
		}
		
		resolve_title_and_vassal_change = scope:change

		scope:new_title = {
			set_coa = scope:old_title
			set_color_from_title = scope:old_title
			set_capital_county = scope:old_title.title_capital_county
		}
		set_primary_title_to = scope:new_title
		
		trigger_event = major_decisions.1101 #Make event for new dynasty
	}
}
